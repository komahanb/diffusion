\begin{homeworkProblem}[4]

  \emph{The following ODE system:
    $$
    {y_1}^\prime = \alpha - y_1 - \dfrac{4y_1 y_2}{1 + y_1^2},
    $$
    $$
    {y_2}^\prime = \beta y_1 \left(1 - \dfrac{y_2}{1+y_1^2} \right),
    $$ where $\alpha$ and $\beta$ are parameters, represents a
    simplified approximation to a chemical reaction. There is a
    parameter value $\beta_c = \frac{3\alpha}{5} - \frac{25}{\alpha}$
    such that for $\beta > \beta_c$ solution trajectories decay in
    amplitude and spiral in phase space into a stable fixed point,
    whereas for $\beta < \beta_c$ trajectories oscillate without
    damping and are attracted to a stable limit cycle. This is called
    a Hopf bifurcation.}

  \part  \emph{Set $\alpha=10$ and use any of the discretization methods
    introduced in this chapter with a fixed step size $h=0.01$ to
    approximate the solution starting at $y_1(0),~y_2(0)=2$, for $0
    \le t \le 20$. Do this for the parameter values $\beta=2$ and
    $\beta = 4$. For each case plot $y_1$ vs. $t$ and $y_2$
    vs. $y_1$. Describe your observations. }

  \part \emph{Investigate the situtation closer to the critical value
    $\beta_c=3.5$. You may have to increase the length of the
    integration interval $b$ to get a better look.}
    
\end{homeworkProblem}
